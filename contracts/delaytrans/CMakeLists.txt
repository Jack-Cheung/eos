file(GLOB ABI_FILES "*.abi")
set(MY_CONTRACT_NAME "delaytrans")
configure_file("${ABI_FILES}" "${CMAKE_CURRENT_BINARY_DIR}" COPYONLY)
#you need to install eosio.cdt 
execute_process(
  COMMAND eosio-cpp 
  ${CMAKE_CURRENT_SOURCE_DIR}/${MY_CONTRACT_NAME}.cpp
   -o ${CMAKE_CURRENT_BINARY_DIR}/${MY_CONTRACT_NAME}.wasm  
  WORKING_DIRECTORY "${CMAKE_CURRENT_SOURCE_DIR}"
  OUTPUT_VARIABLE "${MY_CONTRACT_NAME}_Res"
  ERROR_QUIET
  OUTPUT_STRIP_TRAILING_WHITESPACE)
