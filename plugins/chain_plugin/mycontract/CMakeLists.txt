set(MY_CONTRACT_NAME "mycontract")
#you need to install eosio.cdt 
execute_process(
  COMMAND eosio-cpp 
  ${CMAKE_CURRENT_SOURCE_DIR}/${MY_CONTRACT_NAME}.cpp
   -o ${CMAKE_CURRENT_SOURCE_DIR}/${MY_CONTRACT_NAME}.wasm
  WORKING_DIRECTORY "${CMAKE_CURRENT_SOURCE_DIR}"
  OUTPUT_VARIABLE "${MY_CONTRACT_NAME}_Res"
  ERROR_QUIET
  OUTPUT_STRIP_TRAILING_WHITESPACE)
